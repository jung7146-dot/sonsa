#!/usr/bin/env python3 """ GitHub Pages 자동 업로드 스크립트 - 이 파일을 blankcards.html 과 같은 폴더에 놓고 실행하세요 - Python 3 설치 필요 (https://python.org) - 더블클릭 또는 터미널에서 python github_upload.py """ import base64, json, urllib.request, os, sys TOKEN = "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" REPO = "jung7146-dot/sonsa" FILES = { "blankcards.html": "blankcards.html", # 로컬파일명: GitHub파일명 "index.html": "index.html", } def get_sha(path): req = urllib.request.Request( f"https://api.github.com/repos/{REPO}/contents/{path}", headers={"Authorization": f"token {TOKEN}"} ) try: return json.loads(urllib.request.urlopen(req).read())['sha'] except: return None def upload(local, remote): if not os.path.exists(local): print(f" ⚠ 파일 없음: {local}") return with open(local, 'rb') as f: content = base64.b64encode(f.read()).decode() sha = get_sha(remote) data = {"message": f"Update {remote}", "content": content} if sha: data["sha"] = sha req = urllib.request.Request( f"https://api.github.com/repos/{REPO}/contents/{remote}", data=json.dumps(data).encode(), headers={"Authorization": f"token {TOKEN}", "Content-Type": "application/json"}, method="PUT" ) try: urllib.request.urlopen(req) print(f" ✓ {remote} 업로드 완료") except Exception as e: print(f" ✗ {remote} 실패: {e}") print("=" * 40) print(" GitHub Pages 업로드") print("=" * 40) script_dir = os.path.dirname(os.path.abspath(__file__)) os.chdir(script_dir) for local, remote in FILES.items(): if os.path.exists(local): print(f"\n업로드 중: {local}") upload(local, remote) else: print(f"\n건너뜀 (파일 없음): {local}") print("\n완료! https://jung7146-dot.github.io/sonsa/blankcards.html") input("\nEnter 키를 눌러 닫기...")
